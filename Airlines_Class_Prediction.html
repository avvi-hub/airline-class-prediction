def get_user_input():
    user_data = {}
    user_data['Type of Travel'] = input("Enter Type of Travel ('0'(Business)/'1'(Personal): ")
    user_data['satisfaction'] = input("Enter satisfaction level ('1'(satisfied)/'0'(neutral or dissatisfied): ")
    user_data['Customer Type'] = input("Enter Customer Type ('0'(Loyal)/'1'(Disloyal): ")
    user_data['Flight Distance'] = float(input("Enter Flight Distance: "))
    user_data['Inflight service'] = int(input("Rate Inflight Service (0-5): "))
    user_data['Inflight wifi service'] = int(input("Rate Inflight Wifi Service (0-5): "))
    user_data['Baggage handling'] = int(input("Rate Baggage Handling (0-5): "))
    user_data['Gender'] = input("Enter Gender ('1'(Male)/'0'(Female)): ")
    user_data['Online boarding'] = int(input("Rate Online Boarding (0-5): "))
    user_data['Departure/Arrival time convenient'] = int(input("Rate Time Convenience (0-5): "))
    return pd.DataFrame([user_data])

input_df = get_user_input()
